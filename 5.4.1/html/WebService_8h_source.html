55 static size_t WriteInMemoryCallback(void *contents, size_t size, size_t nmemb, void *userp)
56 {
57  size_t realsize = size * nmemb;
58  struct MemoryStruct *mem = (struct MemoryStruct *)userp;
59 
60  mem->memory = (uint8_t*)realloc(mem->memory, mem->size + realsize + 1);
61  if(mem->memory == NULL) {
62  /* out of memory! */
63  BOOST_LOG_TRIVIAL(error) << "not enough memory (realloc returned NULL)\n";
64  return 0;
65  }
66 
67  memcpy(&(mem->memory[mem->size]), contents, realsize);
68  mem->size += realsize;
69  mem->memory[mem->size] = 0;
70 
71  return realsize;
72 }
