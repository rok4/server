94 Attribution ( json11::Json doc ) : ResourceLocator (), logo(NULL) {
95
96 if (! doc["title"].is_string()) {
97 missing_field = "title";
98 return;
99 }
100 title = doc["title"].string_value();
101
102 if (! doc["url"].is_string()) {
103 missing_field = "url";
104 return;
105 }
106 href = doc["url"].string_value();
107
108 if (doc["logo"].is_object()) {
109 if (! doc["logo"]["width"].is_number()) {
110 missing_field = "logo.width";
111 return;
112 }
113 width = doc["logo"]["width"].number_value();
114 if (! doc["logo"]["height"].is_number()) {
115 missing_field = "logo.height";
116 return;
117 }
118 height = doc["logo"]["height"].number_value();
119
120 if (! doc["logo"]["format"].is_string()) {
121 missing_field = "logo.format";
122 return;
123 }
124 if (! doc["logo"]["url"].is_string()) {
125 missing_field = "logo.url";
126 return;
127 }
128 logo = new ResourceLocator(doc["logo"]["format"].string_value(), doc["logo"]["url"].string_value());
129 }
130 };

138 void add_node_tms(ptree& parent) {
139 ptree& node = parent.add("Attribution", "");
140 node.add("<xmlattr>.href", href);
141 node.add("Title", title);
142
143 if (logo != NULL) {
144 node.add("Logo.<xmlattr>.width", width);
145 node.add("Logo.<xmlattr>.height", height);
146 node.add("Logo.<xmlattr>.href", logo->get_href());
147 node.add("Logo.<xmlattr>.mime-type", logo->get_format());
148 }
149 }
150
157 void add_node_wms(ptree& parent) {
158 ptree& node = parent.add("Attribution", "");
159 node.add("Title", title);
160 node.add("OnlineResource.<xmlattr>.xlink:href", href);
161 node.add("OnlineResource.<xmlattr>.xlink:type", "simple");
162
163 if (logo != NULL) {
164 node.add("LogoURL.<xmlattr>.width", width);
165 node.add("LogoURL.<xmlattr>.height", height);
166 node.add("LogoURL.Format", logo->get_format());
167 node.add("LogoURL.OnlineResource.<xmlattr>.xlink:href", logo->get_href());
168 node.add("LogoURL.OnlineResource.<xmlattr>.xlink:type", "simple");
169 }
170 }
171
178 virtual ~Attribution() {
179 if (logo != NULL) delete logo;
180 };
