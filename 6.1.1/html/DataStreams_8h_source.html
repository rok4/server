128  MessageDataStream ( std::string m, std::string t, int s ) : message ( m ), type ( t ), http_status(s), pos ( 0 ) {}
129 
130  size_t read ( uint8_t *buffer, size_t size ) {
131  if ( size > message.length() - pos ) size = message.length() - pos;
132  memcpy ( buffer, ( uint8_t* ) ( message.data() +pos ),size );
133  pos+=size;
134  return size;
135  }
136  bool eof() {
137  return ( pos==message.length() );
138  }
139  std::string get_type() {
140  return type;
141  }

143  return "";
144  }
145  int get_http_status() {
146  return http_status;
147  }

150  }
