66static DataStream* get_tile(ServicesConfiguration* services, Layer* layer, TileMatrixSet* tms, TileMatrix* tm, int column, int row, std::string format, Style* style) {
67 // Traitement de la requête
68
69 if (tms->get_id() == layer->get_pyramid()->get_tms()->get_id()) {
70 // TMS d'interrogation natif
71 Level* level = layer->get_pyramid()->get_level(tm->get_id());
72
73 DataSource* d = level->get_tile(column, row);
74 if (d == NULL) {
75 return NULL;
76 }
77
78 if (layer->is_raster() && layer->get_pyramid()->get_channels() == 1 && format == "image/png" && style->get_palette() && !style->get_palette()->is_empty()) {
79 return new DataStreamFromDataSource(new PaletteDataSource(d, style->get_palette()));
80 } else {
81 return new DataStreamFromDataSource(d);
82 }
83 } else {
84 // TMS d'interrogation à la demande, forcément du raster
85
86 BoundingBox<double> bbox = tm->tile_indices_to_bbox(column, row);
87 int height = tm->get_tile_height();
88 int width = tm->get_tile_width();
89 CRS* crs = tms->get_crs();
90 bbox.crs = crs->get_request_code();
91
92 bool crs_equals = services->are_crs_equals(layer->get_pyramid()->get_tms()->get_crs()->get_proj_code(), crs->get_proj_code());
93
94 // On se donne maxium 3 tuiles sur 3 dans la pyramide source pour calculer cette tuile
95 Image* image = layer->get_pyramid()->getbbox(3, 3, bbox, width, height, crs, crs_equals, layer->get_resampling(), 0);
96
97 if (image == NULL) {
98 BOOST_LOG_TRIVIAL(warning) << "Cannot process the tile in a non native TMS";
99 return NULL;
100 }
101
102 image->set_bbox(bbox);
103 image->set_crs(crs);
104
105 if (format == "image/png" || format == "png") {
106 return new PNGEncoder(image, style->get_palette());
107 } else if (format == "image/tiff" || format == "image/geotiff") {
108 bool is_geotiff = (format == "image/geotiff");
109
110 // Dans le cas d'un geotiff, on renseigne la valeur de nodata
111 // on ne peut mettre qu'une valeur, ce sera celle du premier canal
112 int nodata = *(layer->get_pyramid()->get_nodata_value());
113
114 // La donnée ne peut être retournée que dans le format de la pyramide source utilisée
115
116 switch (layer->get_pyramid()->get_format()) {
117 case Rok4Format::TIFF_RAW_UINT8:
118 return new TiffRawEncoder<uint8_t>(image, is_geotiff, nodata);
119 case Rok4Format::TIFF_LZW_UINT8:
120 return new TiffLZWEncoder<uint8_t>(image, is_geotiff, nodata);
121 case Rok4Format::TIFF_ZIP_UINT8:
122 return new TiffDeflateEncoder<uint8_t>(image, is_geotiff, nodata);
123 case Rok4Format::TIFF_PKB_UINT8:
124 return new TiffPackBitsEncoder<uint8_t>(image, is_geotiff, nodata);
125 case Rok4Format::TIFF_RAW_FLOAT32:
126 return new TiffRawEncoder<float>(image, is_geotiff, nodata);
127 case Rok4Format::TIFF_LZW_FLOAT32:
128 return new TiffLZWEncoder<float>(image, is_geotiff, nodata);
129 case Rok4Format::TIFF_ZIP_FLOAT32:
130 return new TiffDeflateEncoder<float>(image, is_geotiff, nodata);
131 case Rok4Format::TIFF_PKB_FLOAT32:
132 return new TiffPackBitsEncoder<float>(image, is_geotiff, nodata);
133 default:
134 delete image;
135 return NULL;
136 }
137 } else if (format == "image/jpeg") {
138 switch (layer->get_pyramid()->get_format()) {
139 case Rok4Format::TIFF_JPG_UINT8:
140 return new JPEGEncoder(image, 75);
141 case Rok4Format::TIFF_JPG90_UINT8:
142 return new JPEGEncoder(image, 90);
143 default:
144 delete image;
145 return NULL;
146 }
147
148 } else if (format == "image/x-bil;bits=32") {
149 return new BilEncoder(image);
150 }
151 }
152
153 BOOST_LOG_TRIVIAL(error) << "On ne devrait pas passer par là";
154 return NULL;
155}
